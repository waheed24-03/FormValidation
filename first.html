const express = require('express'); const app = express(); const port = 3000; const path = require('path'); app.use(express.static('public')); app.get('/', (req, res) => { res.sendFile(path.join(__dirname, 'public', 'index.html')); }); app.get('/user/:name', (req, res) => { const userName = req.params.name; res.send(`Hello, ${userName}!`); }); app.get('/search', (req, res) => { const searchQuery = req.query.query; res.send(`Searching for: ${searchQuery}`); }); app.listen(port, () => { console.log(`Server running at http://localhost:${port}`); });
